pipeline {
    agent any

    parameters {
            string(name: 'Username', defaultValue: 'Ross', description: 'Who is running the build')
        }
    stages {    
        stage('Build') {
            steps {
                echo "Hello ${params.Username}, we are building $BUILD_TAG"
                sh './gradlew build'
                echo "Build Success"
            }
            post {
                failure {
                    echo 'FAILURE IN BUILD $BUILD_NUMBER'
                }
            }
        }
        stage('Test') {
            steps {
                parallel (
                    'test 1': {
                        echo "running $BUILD_NUMBER"
                        sh './gradlew run --info'
                    },
                    'test 2': {
                        echo 'running test 2'
                    }
                )
            }
            post {
                failure {
                    echo 'FAILURE WHEN RUNNING BUILD $BUILD_NUMBER'
                }
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
            post {
                failure {
                    echo 'FAILURE IN DEPLOYMENT'
                }
            }
        }
    }
}
